public class StringReverseRecursive {
    public static String reverseString(String str) {
        if (str == null || str.isEmpty() || str.length() == 1) {
            return str;
        } else {
            return reverseString(str.substring(1)) + str.charAt(0);
        }
    }

    public static void main(String[] args) {
        String originalString1 = "hello";
        String reversedString1 = reverseString(originalString1);
        System.out.println("Original: \"" + originalString1 + "\", Reversed: \"" + reversedString1 + "\"");

    }
}

public class Demo {
    static int findMax(int[] arr) {
        int max = arr[0];
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] > max) {
                max = arr[i];
            }
        }
        return max;
    }
    
    public static void main(String[] args) {
        int[] numbers = {23, 45, 12, 67, 34};
        int result = findMax(numbers);
        System.out.println("Maximum: " + result);
    }
}

    Scanner sc = new Scanner(System.in);

    int x= n;

            System.out.print(x + " is not a circular prime"); 
            return;
        }
        String s = Integer.toString(n);
        char ch = s.charAt(0);
        s = s.substring(1);
        s += ch +"";
        n = Integer.parseInt(s);
        
    }
    System.out.print(x + " is a circular prime");
    }
    public static boolean isPrime(int n){
        for(int i = 2; i <= n/ 2; i++){
            if(n % i == 0){
                return false;
            }
        }
        return true;
    }
}

public class RightShift{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int arr[] = new int[n];
        for(int  i = 0; i < n; i++){
            arr[i] = sc.nextInt();
        }
        int k = sc.nextInt();
        k %= n;
        reverse(arr, 0, arr.length - k -1);
        reverse(arr,arr.length - k, arr.length - 1);
        reverse(arr, 0, arr.length - 1);
        System.out.print(Arrays.toString(arr));
    }
    public static void reverse(int arr[], int start, int end){
        for(int i = start, j = end; i < j; i++,j--){
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
}
